Unit 4: Object-Oriented Programming

This unit focused on using classes and objects to organize larger programs. My summative project was a Character Battle Arena where characters have stats, actions, and battle behaviours.

OOP Principles

In my Character Battle Arena, I used encapsulation by creating a Character class that stores each character's name, attack, defense, health, and speed. Instead of directly changing these values throughout the program, I used methods such as getHealth(), takeDamage(), heal(), and attackTarget(). This keeps the character data and behaviours together, making the program easier to understand, test, and update.

The project also shows modular design because the class logic is separated from the main game logic. The character.js file defines what a character is and what it can do, while main.js handles user commands, loading rosters, drawing the canvas, and running the battle. This separation makes the code more maintainable because changes to character behaviour do not require rewriting the entire game.

Copyable Code: character.js

// character.js
// Character class for the Character Battle Arena summative.
// This file demonstrates encapsulation because each Character object stores
// its own name, attack, defense, health, speed, and battle behaviours.

class Character {
  constructor(name, attack, defense, health, speed) {
    this.name = name;
    this.attack = Number(attack);
    this.defense = Number(defense);
    this.health = Number(health); // TODO #4 fixed: getHealth() returns this variable.
    this.maxHealth = Number(health);
    this.speed = Number(speed);
  }

  getName() {
    return this.name;
  }

  getHealth() {
    return this.health;
  }

  getAttack() {
    return this.attack;
  }

  getDefense() {
    return this.defense;
  }

  getSpeed() {
    return this.speed;
  }

  isAlive() {
    return this.getHealth() > 0; // TODO #12 fixed: getHealth() is called with ().
  }

  takeDamage(amount) {
    this.health = Math.max(0, this.health - amount);
  }

  attackTarget(target) {
    const baseDamage = this.getAttack() - Math.floor(target.getDefense() / 2);
    const damage = Math.max(5, baseDamage);
    target.takeDamage(damage);
    return damage;
  }

  heal() {
    const healAmount = 20;
    this.health = Math.min(this.maxHealth, this.health + healAmount);
    return healAmount;
  }
}

// main.js
// Main game logic for Character Battle Arena.
// This version corrects the teacher feedback from TODO #1, #4, #7, #11, #13, and #14.

const homeRosterText = `Pikachu 55 40 35 90
Charizard 84 78 78 100
Bulbasaur 49 49 45 45
Squirtle 48 65 44 43
Gengar 65 60 60 110
Mewtwo 110 90 106 130`;

const enemyRosterText = `Snorlax 110 65 160 30
Dragonite 134 95 91 80
Lapras 85 80 130 60
Machamp 130 80 90 55
Alakazam 50 45 55 120
Gyarados 125 79 95 81`;

let homeCharacters = [];
let enemyCharacters = [];
let hChar = null;
let eChar = null;
let battleStarted = false;

const logBox = document.getElementById("battle-log");
const commandInput = document.getElementById("command-input");
const movesContainer = document.getElementById("moves-container");
const canvas = document.getElementById("gameCanvas");
const ctx = canvas.getContext("2d");

function log(message) {
  const line = document.createElement("div");
  line.textContent = "> " + message;
  logBox.appendChild(line);
  logBox.scrollTop = logBox.scrollHeight;
}

function parseRosterLine(line) {
  const data = line.trim().split(/\s+/);
  const name = data[0];

  // TODO #11 fixed: string data is parsed into numeric data before creating objects.
  const attack = parseInt(data[1]);
  const defense = parseInt(data[2]);
  const health = parseInt(data[3]);
  const speed = parseInt(data[4]);

  return new Character(name, attack, defense, health, speed);
}

function loadCharacters() {
  homeCharacters = homeRosterText.trim().split("\n").map(line => {
    // TODO #1 fixed: character object is declared with let.
    let character = parseRosterLine(line);
    return character;
  });

  enemyCharacters = enemyRosterText.trim().split("\n").map(line => {
    // TODO #4 and #7 fixed: character object is declared with let.
    let character = parseRosterLine(line);
    return character;
  });

  hChar = homeCharacters[0];
  eChar = enemyCharacters[0];
  battleStarted = false;
  log("Characters loaded successfully.");
  log(`Home character selected: ${hChar.getName()}`);
  log(`Enemy character selected: ${eChar.getName()}`);
  drawBattle();
}

function viewCharacters() {
  if (homeCharacters.length === 0 || enemyCharacters.length === 0) {
    log("Load characters first.");
    return;
  }

  log("HOME ROSTER:");
  homeCharacters.forEach((character, index) => {
    log(`${index + 1}. ${character.getName()} | HP ${character.getHealth()} | ATK ${character.getAttack()} | DEF ${character.getDefense()} | SPD ${character.getSpeed()}`);
  });

  log("ENEMY ROSTER:");
  enemyCharacters.forEach((character, index) => {
    log(`${index + 1}. ${character.getName()} | HP ${character.getHealth()} | ATK ${character.getAttack()} | DEF ${character.getDefense()} | SPD ${character.getSpeed()}`);
  });
}

function startBattle() {
  if (!hChar || !eChar) {
    log("Load characters before starting the battle.");
    return;
  }

  battleStarted = true;
  movesContainer.innerHTML = "";

  const attackButton = document.createElement("button");
  attackButton.textContent = "Attack";
  attackButton.addEventListener("click", playerAttack);

  const healButton = document.createElement("button");
  healButton.textContent = "Heal";
  healButton.addEventListener("click", playerHeal);

  movesContainer.appendChild(attackButton);
  movesContainer.appendChild(healButton);

  log("Battle started!");
  drawBattle();
}

function playerAttack() {
  if (!battleStarted) return;

  // TODO #13 and #14 fixed: use eChar consistently instead of characterEnemy.
  const damage = hChar.attackTarget(eChar);
  log(`${hChar.getName()} attacked ${eChar.getName()} for ${damage} damage.`);

  if (!eChar.isAlive()) {
    log(`${eChar.getName()} fainted. You win!`);
    battleStarted = false;
    drawBattle();
    return;
  }

  enemyTurn();
  drawBattle();
}

function playerHeal() {
  if (!battleStarted) return;
  const healAmount = hChar.heal();
  log(`${hChar.getName()} healed for ${healAmount} HP.`);
  enemyTurn();
  drawBattle();
}

function enemyTurn() {
  if (!eChar.isAlive()) return;

  const damage = eChar.attackTarget(hChar);
  log(`${eChar.getName()} attacked ${hChar.getName()} for ${damage} damage.`);

  if (!hChar.isAlive()) {
    log(`${hChar.getName()} fainted. You lost the battle.`);
    battleStarted = false;
  }
}

function drawCharacter(character, x, y, color) {
  ctx.fillStyle = color;
  ctx.fillRect(x, y, 90, 90);
  ctx.fillStyle = "#111";
  ctx.font = "16px Courier New";
  ctx.fillText(character.getName(), x - 5, y - 15);
  ctx.fillText(`HP: ${character.getHealth()}`, x, y + 115);
}

function drawBattle() {
  ctx.clearRect(0, 0, canvas.width, canvas.height);
  ctx.fillStyle = "#e0f7fa";
  ctx.fillRect(0, 0, canvas.width, canvas.height);

  if (hChar) drawCharacter(hChar, 80, 150, "#4caf50");
  if (eChar) drawCharacter(eChar, 330, 60, "#ff5252");
}

commandInput.addEventListener("keydown", event => {
  if (event.key !== "Enter") return;

  const command = commandInput.value.trim();
  commandInput.value = "";

  if (command === "1") loadCharacters();
  else if (command === "2") log("Create Character feature is not active in this embedded demo.");
  else if (command === "3") viewCharacters();
  else if (command === "4") log("Edit Character feature is not active in this embedded demo.");
  else if (command === "5") startBattle();
  else log("Invalid command. Use 1, 2, 3, 4, or 5.");
});

drawBattle();

Summary: Why OOP Helps Larger Projects

Modular design with OOP is more efficient for large-scale projects than procedural programming because it separates the program into smaller, reusable parts. In this project, the Character class acts like a blueprint for every character, so I do not need to rewrite the same health, attack, and damage logic for each new character. This makes the code easier to expand because I can add more characters or new abilities while reusing the same structure. It also makes debugging easier because if there is an issue with character health or damage, I know to check the class methods instead of searching through one long procedural file.
